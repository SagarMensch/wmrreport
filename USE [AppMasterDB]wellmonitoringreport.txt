CREATE   VIEW [dbo].[WellMonitoringReport]
AS
WITH wk AS (
    SELECT DISTINCT Week_Number
    FROM dbo.WMR
),
target_week AS (
    SELECT Week_Number
    FROM wk
    ORDER BY Week_Number DESC
    OFFSET 1 ROW FETCH NEXT 1 ROW ONLY
),
latest AS (
    SELECT w.*
    FROM dbo.WMR w
    INNER JOIN target_week tw
        ON w.Week_Number = tw.Week_Number
),

/* ---------- TaskPlan: parse + aggregate per well ---------- */
taskplan_parsed AS (
    SELECT
        tp.New_Well_ID,
        LTRIM(RTRIM(tp.[text])) AS tp_text,
        UPPER(LTRIM(RTRIM(tp.[text]))) AS tp_text_u,
        LTRIM(RTRIM(tp.[code])) AS tp_code,
        dbo.fn_WMR_ParseProgress(CAST(tp.[progress] AS NVARCHAR(510))) AS tp_prog,
        dbo.fn_WMR_ParseDate(CAST(tp.[actual_start] AS NVARCHAR(510))) AS tp_start_d,
        dbo.fn_WMR_ParseDate(CAST(tp.[actual_end] AS NVARCHAR(510))) AS tp_end_d,
        CASE
            WHEN tp.[code] IS NOT NULL AND LTRIM(RTRIM(tp.[code])) LIKE 'FLME%'
                THEN TRY_CONVERT(INT, SUBSTRING(LTRIM(RTRIM(tp.[code])), 5, 4))
            ELSE NULL
        END AS flme_no
    FROM dbo.WMR_TaskPlan_csv_imported tp
),
taskplan_agg AS (
    SELECT
        p.New_Well_ID,
        MAX(CASE WHEN p.tp_text = 'Location' THEN p.tp_prog END) AS tp_location_progress,
        MAX(CASE WHEN p.tp_text = 'Location Start' THEN p.tp_start_d END) AS tp_location_start_date,
        MAX(CASE WHEN p.tp_text_u = 'LOCATION FINISH' THEN p.tp_end_d END) AS tp_location_finish_date,
        MAX(CASE WHEN p.flme_no BETWEEN 1360 AND 1430 THEN p.tp_end_d END) AS tp_flme_const_complete_date,
        MAX(CASE WHEN p.tp_text = 'Flowline Completion' THEN p.tp_prog END) AS tp_flowline_completion_progress,
        MAX(CASE WHEN p.tp_code LIKE '%LCPC1005%' THEN p.tp_prog END) AS tp_access_road_5,
        MAX(CASE WHEN p.tp_code LIKE '%LCPC1040%' THEN p.tp_prog END) AS tp_earth_work_60,
        MAX(CASE WHEN p.tp_code LIKE '%LCCC1040%' THEN p.tp_prog END) AS tp_cellar_20,
        MAX(CASE WHEN p.tp_code LIKE '%LCCC1070%' THEN p.tp_prog END) AS tp_beam_pump_base_5,
        MAX(CASE WHEN p.tp_code LIKE '%LCCC1080%' OR p.tp_code LIKE '%LCCC1090%' THEN p.tp_prog END) AS tp_earthing_septic_max,
        MAX(CASE WHEN p.tp_code LIKE '%LCPC1030%' THEN p.tp_prog END) AS tp_water_waste_2,
        MAX(CASE WHEN p.tp_code LIKE '%LCPC1050%' THEN p.tp_prog END) AS tp_hdpe_liner_4
    FROM taskplan_parsed p
    GROUP BY p.New_Well_ID
),
taskplan_lcpc1050 AS (
    SELECT
        p.New_Well_ID,
        MAX(p.tp_prog) AS lcpc1050_prog
    FROM taskplan_parsed p
    WHERE p.tp_code LIKE '%LCPC1050%'
    GROUP BY p.New_Well_ID
),

/* ---------- Design Tracker ----------
   E_I_AFC_Completion_Date may contain multiple dates in one cell.
   Pick the latest valid parsed date for each well.
*/
designtracker_tokens AS (
    SELECT
        LTRIM(RTRIM(d.Well_ID)) AS Well_ID,
        dbo.fn_WMR_ParseDate(
            LTRIM(RTRIM(
                CASE
                    WHEN CHARINDEX('(', s.value) > 0
                        THEN LEFT(s.value, CHARINDEX('(', s.value) - 1)
                    ELSE s.value
                END
            ))
        ) AS afc_date_piece,
        dbo.fn_WMR_ParseDate(CAST(d.Survey_Received_Date AS NVARCHAR(510))) AS survey_received_date_piece
    FROM dbo.DesignTrackerCSVImport d
    OUTER APPLY STRING_SPLIT(
        REPLACE(REPLACE(REPLACE(CAST(d.E_I_AFC_Completion_Date AS NVARCHAR(MAX)), CHAR(13), CHAR(10)), CHAR(10) + CHAR(10), CHAR(10)), ';', CHAR(10)),
        CHAR(10)
    ) s
),
designtracker_flag AS (
    SELECT
        t.Well_ID,
        MAX(t.afc_date_piece) AS engg_finish_date_d,
        MAX(t.survey_received_date_piece) AS survey_received_date_d,
        CASE
            WHEN MAX(t.afc_date_piece) IS NOT NULL THEN CAST(1.0 AS DECIMAL(10,4))
            ELSE CAST(0.0 AS DECIMAL(10,4))
        END AS engg_done_flag
    FROM designtracker_tokens t
    GROUP BY t.Well_ID
),

/* ---------- Well Type override ---------- */
welltype_lookup AS (
    SELECT
        LTRIM(RTRIM(wt.Well_ID)) AS Well_ID,
        NULLIF(LTRIM(RTRIM(wt.Lift_type)), '') AS Lift_type
    FROM dbo.[2026_Well_Delivery_Scope_Well_Type] wt
),

/* ---------- Bulk Update join ---------- */

        bu.sap_notification_received_date_z6_2_days_before_eng_completion AS bu_sap_notification_received_date_z6_2_days_before_eng_completion,
        bu.actual_rig_on_date AS bu_actual_rig_on_date,
        bu.actual_rig_off_date AS bu_actual_rig_off_date,
        bu.wlctf_acceptanceapproval_from_production AS bu_wlctf_acceptanceapproval_from_production,
        bu.actual_hoist_fbu_rsr_on_date AS bu_actual_hoist_fbu_rsr_on_date,
        bu.actual_hoist_fbu_rsr_off_date AS bu_actual_hoist_fbu_rsr_off_date,
        bu.wellpad_handover_2_from_hoist_fbu_rsr_off_date AS bu_wellpad_handover_2_from_hoist_fbu_rsr_off_date,
        bu.actual_eng_completion_date AS bu_actual_eng_completion_date,
        bu.actual_comm_start_date AS bu_actual_comm_start_date,
        bu.actual_comm_finish_date_with_in_2_days_from_actual_engg_completion_date AS bu_actual_comm_finish_date_with_in_2_days_from_actual_engg_completion_date,
        bu.actual_start_date AS bu_actual_start_date,
        bu.actual_finish_date AS bu_actual_finish_date,
        bu.const_actual_start_date AS bu_const_actual_start_date,
        bu.const_complete_date_including_f_l_final_hydro_test_1_day_before_rig_on_date AS bu_const_complete_date_including_f_l_final_hydro_test_1_day_before_rig_on_date
    FROM latest w
    LEFT JOIN dbo.WMR_SQL_Bulk_Update bu
        ON bu.rig_no = w.rig_no
       AND bu.pdo_well_id = w.pdo_well_id
),

typed AS (
    SELECT
        j.*,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_z6_data_submitted_date), dbo.fn_WMR_ParseDate(j.z6_data_submitted_date)) AS z6_eff_d,
        COALESCE(
            dbo.fn_WMR_ParseDate(j.bu_sap_notification_received_date_z6_2_days_before_eng_completion),
            dbo.fn_WMR_ParseDate(j.[sap_notification_received_date_z6_2_days_before_eng._completion])
        ) AS sap_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_rig_on_date), dbo.fn_WMR_ParseDate(j.actual_rig_on_date)) AS rig_on_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_rig_off_date), dbo.fn_WMR_ParseDate(j.actual_rig_off_date)) AS rig_off_eff_d,
        COALESCE(
            dbo.fn_WMR_ParseDate(j.bu_wlctf_acceptanceapproval_from_production),
            dbo.fn_WMR_ParseDate(j.wlctf_acceptanceapproval_from_production)
        ) AS wlctf_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_hoist_fbu_rsr_on_date), dbo.fn_WMR_ParseDate(j.actual_hoist_fbu_rsr_on_date)) AS hoist_on_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_hoist_fbu_rsr_off_date), dbo.fn_WMR_ParseDate(j.actual_hoist_fbu_rsr_off_date)) AS hoist_off_eff_d,
        COALESCE(
            dbo.fn_WMR_ParseDate(j.bu_wellpad_handover_2_from_hoist_fbu_rsr_off_date),
            dbo.fn_WMR_ParseDate(j.[wellpad_handover-2_from_hoist_fbu_rsr_off_date])
        ) AS handover2_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_eng_completion_date), dbo.fn_WMR_ParseDate(j.[actual_eng._completion_date])) AS eng_comp_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_comm_start_date), dbo.fn_WMR_ParseDate(j.[actual_comm._start_date])) AS comm_start_eff_d,
        COALESCE(
            dbo.fn_WMR_ParseDate(j.bu_actual_comm_finish_date_with_in_2_days_from_actual_engg_completion_date),
            dbo.fn_WMR_ParseDate(j.[actual_comm._finish_date_with_in_2_days_from_actual_engg._completion_date])
        ) AS comm_finish_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_start_date), dbo.fn_WMR_ParseDate(j.actual_start_date)) AS actual_start_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_actual_finish_date), dbo.fn_WMR_ParseDate(j.actual_finish_date)) AS actual_finish_eff_d,
        COALESCE(dbo.fn_WMR_ParseDate(j.bu_const_actual_start_date), dbo.fn_WMR_ParseDate(j.[const._actual_start_date])) AS const_start_eff_d,
        COALESCE(
            dbo.fn_WMR_ParseDate(j.bu_const_complete_date_including_f_l_final_hydro_test_1_day_before_rig_on_date),
            dbo.fn_WMR_ParseDate(j.[const._complete_date_including_f_l_final_hydro_test_1_day_before_rig_on_date])
        ) AS const_finish_eff_d
    FROM joined j
),

/* ---------- Score rows ---------- */
scored AS (
    SELECT
        t.*,
        (
            SELECT COUNT(*)

                (t.sl_no),
                (t.rig_no),
                (t.well_location),
                (t.well_name_after_spud),
                (t.pdo_well_id),

                (t.flow_linedl),
                (t.location_po_no),
                (t.location_po_recvd_date),
                (t.scr_no),
                (t.scr_date),
                (t.moc_raised),
                (t.moc_approved),
                (t.actual_pegged_date),
                (t.last_week_cum_progress),
                (t.cum_progress_for_this_week),
                (t.actual_start_date),
                (t.actual_finish_date),
                (t.flaf_issue_date),
                (t.ramz_id),
                (t.flow_line_po_no),
                (t.[f_l_po_recd._date]),
                (t.station_name_no),
                (t.physical_tie_in_port_number),
                (t.physical_tie_in_port_available_when_flaf_issued),
                (t.fl_dia),
                (t.fl_length_meter),
                (t.ohl_length_meter),
                (t.ohl_progress),
                (t.ohl_completion_date),
                (t.actual_rig_on_date),
                (t.actual_rig_off_date),
                (t.wlctf_acceptanceapproval_from_production),
                (t.[actual_comm._start_date]),
                (t.[actual_comm._finish_date_with_in_2_days_from_actual_engg._completion_date]),
                (t.project_id),
                (t.Week_Number)
            ) v(val)
            WHERE v.val IS NOT NULL
              AND LTRIM(RTRIM(CONVERT(NVARCHAR(4000), v.val))) <> ''
              AND LOWER(LTRIM(RTRIM(CONVERT(NVARCHAR(4000), v.val)))) <> 'null'
        ) AS non_empty_count
    FROM typed t
),

ranked AS (
    SELECT
        s.*,
        ROW_NUMBER() OVER(
            PARTITION BY
                UPPER(LTRIM(RTRIM(s.pdo_well_id))),
                UPPER(LTRIM(RTRIM(s.well_location)))
            ORDER BY s.non_empty_count DESC
        ) AS rn_keep
    FROM scored s
    WHERE s.pdo_well_id IS NOT NULL
      AND LTRIM(RTRIM(s.pdo_well_id)) <> ''
      AND LOWER(LTRIM(RTRIM(s.pdo_well_id))) <> 'null'
      AND s.well_location IS NOT NULL
      AND LTRIM(RTRIM(s.well_location)) <> ''
      AND LOWER(LTRIM(RTRIM(s.well_location))) <> 'null'
),

        COALESCE(MAX(CASE WHEN rn_keep = 1 THEN scr_no END), MAX(scr_no)) AS scr_no,
        COALESCE(MAX(CASE WHEN rn_keep = 1 THEN scr_date END), MAX(scr_date)) AS scr_date,

/* ---------- Buffer logic ---------- */
rol_per_rig AS (
    SELECT
        d.rig_no,
        d.pdo_well_id,
        d.rig_on_eff_d,
        ROW_NUMBER() OVER (PARTITION BY d.rig_no ORDER BY d.rig_on_eff_d DESC, d.pdo_well_id) AS rn
    FROM dedup d
    WHERE d.rig_on_eff_d IS NOT NULL
      AND d.rig_off_eff_d IS NULL
),
rol_one AS (
    SELECT rig_no, pdo_well_id
    FROM rol_per_rig
    WHERE rn = 1
),
sap_seq AS (
    SELECT
        s.Work_Center,
        s.Well_ID,
        dbo.fn_WMR_ParseDate(CAST(s.Earl_start_date AS NVARCHAR(510))) AS earl_start_d,
        ROW_NUMBER() OVER (
            PARTITION BY s.Work_Center
            ORDER BY dbo.fn_WMR_ParseDate(CAST(s.Earl_start_date AS NVARCHAR(510))) ASC, s.Well_ID
        ) AS seq_no
    FROM dbo.SAP_DRILLING_SEQUENCE_Staging s
    WHERE s.Work_Center IS NOT NULL AND LTRIM(RTRIM(s.Work_Center)) <> ''
      AND s.Well_ID IS NOT NULL AND LTRIM(RTRIM(s.Well_ID)) <> ''
),
rol_pos AS (
    SELECT
        r.rig_no,
        r.pdo_well_id AS rol_well_id,
        ss.seq_no AS rol_seq_no
    FROM rol_one r
    LEFT JOIN sap_seq ss
        ON ss.Work_Center = r.rig_no
       AND ss.Well_ID = r.pdo_well_id
),
ssfd_lookup AS (
    SELECT DISTINCT
        LTRIM(RTRIM(s.Well_ID)) AS Well_ID,
        'SSFD' AS ssfd_value
    FROM dbo.SAP_DRILLING_SEQUENCE_Staging s
    WHERE UPPER(LTRIM(RTRIM(s.Station_Code))) IN ('SSFD', 'SSFDP')
),
buffer_candidates AS (
    SELECT
        ss.Work_Center AS rig_no,
        ss.Well_ID AS pdo_well_id,
        ss.seq_no
    FROM sap_seq ss
    INNER JOIN rol_pos rp
        ON rp.rig_no = ss.Work_Center
       AND rp.rol_seq_no IS NOT NULL
       AND ss.seq_no > rp.rol_seq_no
    INNER JOIN taskplan_lcpc1050 lc
        ON lc.New_Well_ID = ss.Well_ID
       AND lc.lcpc1050_prog >= 0.96
    INNER JOIN dedup d
        ON d.rig_no = ss.Work_Center
       AND d.pdo_well_id = ss.Well_ID
),
buffer_ranked AS (
    SELECT
        bc.rig_no,
        bc.pdo_well_id,
        ROW_NUMBER() OVER (PARTITION BY bc.rig_no ORDER BY bc.seq_no) AS buffer_n
    FROM buffer_candidates bc
),

computed AS (
    SELECT
        d.*,
        tp.tp_location_progress,
        tp.tp_location_start_date,
        tp.tp_location_finish_date,
        tp.tp_flme_const_complete_date,
        tp.tp_flowline_completion_progress,

        tp.tp_earth_work_60,

        tp.tp_beam_pump_base_5,
        tp.tp_earthing_septic_max,
        tp.tp_water_waste_2,
        tp.tp_hdpe_liner_4,

        COALESCE(dt.engg_done_flag, CAST(0.0 AS DECIMAL(10,4))) AS engg_done_flag,
        dt.survey_received_date_d,
        dt.engg_finish_date_d,
        br.buffer_n,

        COALESCE(wt.Lift_type, d.well_type) AS well_type_override,
        COALESCE(ss.ssfd_value, d.ssfd_wells) AS ssfd_override,

        CASE
            WHEN d.rig_off_eff_d IS NOT NULL THEN 'drilled'
            WHEN d.rig_on_eff_d IS NOT NULL AND d.rig_off_eff_d IS NULL THEN 'ROL'
            ELSE NULL
        END AS buffer_status_base,

        CASE
            WHEN d.rig_off_eff_d IS NOT NULL AND d.rig_on_eff_d IS NOT NULL AND d.rig_off_eff_d < d.rig_on_eff_d THEN 'Error'
            WHEN d.wlctf_eff_d IS NOT NULL AND d.rig_off_eff_d IS NOT NULL AND d.wlctf_eff_d < d.rig_off_eff_d THEN 'Error'
            WHEN d.hoist_on_eff_d IS NOT NULL AND d.wlctf_eff_d IS NOT NULL AND d.hoist_on_eff_d < d.wlctf_eff_d THEN 'Error'
            WHEN d.hoist_off_eff_d IS NOT NULL AND d.wlctf_eff_d IS NOT NULL AND d.hoist_off_eff_d < d.wlctf_eff_d THEN 'Error'
            WHEN d.handover2_eff_d IS NOT NULL AND d.hoist_off_eff_d IS NOT NULL AND d.handover2_eff_d < d.hoist_off_eff_d THEN 'Error'
            WHEN d.eng_comp_eff_d IS NOT NULL AND d.handover2_eff_d IS NOT NULL AND d.eng_comp_eff_d < d.handover2_eff_d THEN 'Error'
            WHEN d.comm_finish_eff_d IS NOT NULL AND d.eng_comp_eff_d IS NOT NULL AND d.comm_finish_eff_d < d.eng_comp_eff_d THEN 'Error'
            WHEN d.comm_finish_eff_d IS NOT NULL AND d.comm_start_eff_d IS NOT NULL AND d.comm_finish_eff_d < d.comm_start_eff_d THEN 'Error'
            ELSE NULL
        END AS data_error_calc,

        CASE
            WHEN d.rig_off_eff_d IS NULL THEN NULL
            ELSE
                CASE
                    WHEN d.comm_start_eff_d IS NOT NULL THEN
                        CASE
                            WHEN d.comm_start_eff_d > DATEADD(DAY, 5, d.rig_off_eff_d)
                                THEN DATEDIFF(DAY, DATEADD(DAY, 5, d.rig_off_eff_d), d.comm_start_eff_d)
                            ELSE 0
                        END
                    ELSE
                        CASE
                            WHEN CAST(GETDATE() AS DATE) > DATEADD(DAY, 5, d.rig_off_eff_d)
                                THEN DATEDIFF(DAY, DATEADD(DAY, 5, d.rig_off_eff_d), CAST(GETDATE() AS DATE))
                            ELSE NULL
                        END
                END
        END AS engg_kpi_calc
    FROM dedup d
    LEFT JOIN taskplan_agg tp
        ON tp.New_Well_ID = d.pdo_well_id
    LEFT JOIN designtracker_flag dt
        ON dt.Well_ID = LTRIM(RTRIM(d.pdo_well_id))
    LEFT JOIN buffer_ranked br
        ON br.rig_no = d.rig_no
       AND br.pdo_well_id = d.pdo_well_id
    LEFT JOIN welltype_lookup wt
        ON wt.Well_ID = LTRIM(RTRIM(d.pdo_well_id))
    LEFT JOIN ssfd_lookup ss
        ON ss.Well_ID = LTRIM(RTRIM(d.pdo_well_id))
)
